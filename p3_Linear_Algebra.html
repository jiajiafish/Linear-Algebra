        1.inftyTrue or False, p 
        2.intfloat
    
"""
def calc_Norm(x, p = 2, infty = False):
    if infty:
        x = np.array(x,dtype=float)
        max_data = np.max(x)
        X = np.abs(max_data)
    elif p ==0:
        X = np.count_nonzero(x)
    else:
        x = np.array(x,dtype=float)
        power_sum = np.sum(np.power(x, p))
        X = np.power(power_sum,1/p)
    return X
